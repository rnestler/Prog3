\subsection{Multiple Inheritance}
\begin{lstlisting}
	class Base {};
	struct MixIn{};

	};
\end{lstlisting}

\begin{itemize}

		members of base are visible and usable from derived class
	\item protected:
		members of base are usable from derived class
	\item private:
		members of base can not be accessed
\end{itemize}
